const merge2Arrays = (firstArray, secondArray) => {
    const sortArray = [];
    let i =0, j = 0;

    while (i < firstArray.length && j < secondArray.length) {
        if (firstArray[i] < secondArray[j]) {
            sortArray.push(firstArray[i]);
            i++;
        } else {
            sortArray.push(secondArray[j]);
            j++;
        }
    }   
    return [...sortArray, 
            ...firstArray.slice(i), 
            ...secondArray.slice(j)];
};
